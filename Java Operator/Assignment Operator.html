/*
Assignment opearators = ( += , -= ,*= , /= , %= )
 */
package opratorassignment;

/**
 *
 * @author Md Mahin Hossain
 */
public class OpratorAssignment {

    public static void main(String[] args) {
       
        int x = 3; 
        int y = 2;
        
        System.out.println("X = 3 \t Y = 2 \n");
 
  
       
        
        
        x+=y; // x = x+y = 5
        System.out.println("X and Y Summation = "+x);
        
        x-=y; // x = x-y = 3
        System.out.println("X and Y Substraction = "+x);
        
        x*=y; // x = x*y = 6
        System.out.println("X and Y Multiplication = "+x);
        
        x/=y; // x = x/y = 3
        System.out.println("X and Y Devidation = "+x);
        
        x%=y; // x = x%y = 1
        System.out.println("X and Y Modulas = "+x);
        
        
        
    }
    
}
