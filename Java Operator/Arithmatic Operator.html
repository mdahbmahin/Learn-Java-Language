 This is Simple calculator code using Arithmatic Operator

/*
 
 Arithmatic opearators = ( +,-,*,/,% )   

 */
package operatorarithmatic;

import java.util.Scanner;
public class OperatorArithmatic {

    
    public static void main(String[] args) {
        
        
        int num1, num2, result;
        
        Scanner input = new Scanner ( System.in);
        
        System.out.print("Enter Your First Number : ");
        num1= input.nextInt();
        System.out.println("Your First Number Is "+num1);
        
        System.out.print("Enter Your Second Number : ");
        num2= input.nextInt();
        System.out.println("Your Second Number is "+num2);
        
        
        
        result = num1 + num2 ;
        System.out.println("summation is :"+result);
        
        result = num1 - num2 ;
        System.out.println("Subtraction is :"+result);
        
        result = num1 * num2 ;
        System.out.println("Multiplication is :"+result);
        
        double  Doubleresult = (double )num1 / num2 ;  //This is typecasting Line .
        System.out.println("Dividation is :"+  Doubleresult);
        
        
        result = num1 % num2 ;
        System.out.println("Remainder is :"+result);
        
  
        
    }
    
}
